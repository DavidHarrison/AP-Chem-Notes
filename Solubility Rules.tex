Rules 1-6 must be applied in order. Earlier rules take precedence.
\begin{enumerate}
	\item Soluble: all ammonium (\ce{NH4+}) and group IA Salts
	\item Soluble: all nitrates (\ce{NO3-}), acetates (\ce{CHCOO-}) and perchlorates (\ce{ClO4-})
	\item Insoluble: all silver (\ce{Ag+}), lead (\ce{Pb+}) and mercury(I) (\ce{Hg2^{2+}}) salts
	\item Soluble: all chlorides (\ce{Cl-}), bromides (\ce{Br-}) and iodides (\ce{I-})
	\item Insoluble: all carbonates (\ce{CO3^{2-}}), chromates (\ce{CrO4^{2-}}), phosphates (\ce{PO4^{3-}}), sulfides (\ce{S^{2-}}), oxides (\ce{O^{2-}}), hydroxides (\ce{OH-})\\
		Exceptions:
		\begin{itemize}
			\item Group II chromates are soluble except \ce{BaCrO4}
			\item Group IIA hydroxides are soluble except \ce{Mg(OH)2}
		\end{itemize}
	\item Soluble: all sulfates, except barium sulfate (\ce{BaSO4}), calcium sulfate (\ce{CaSO4}) and strontium sulfate (\ce{SrO4})
\end{enumerate}
